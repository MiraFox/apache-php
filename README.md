# apache-php
Docker image apache-php

# Доступные переменные PHP:

PHP_TIMEZONE=timezone

PHP_MEMORY_LIMIT=nubmer

PHP_SHORT_OPEN_TAG=On/Off

PHP_POST_MAX_SIZE=number

# Запуск

```
sudo docker run -d mirafox/apache-php
```
